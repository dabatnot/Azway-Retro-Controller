103void drawLogo()
104{
105 display.drawXbm(29, 10, Azway_Logo_width, Azway_Logo_height, Azway_Logo);
106
107 display.setTextAlignment(TEXT_ALIGN_CENTER);
108 display.setFont(ArialMT_Plain_10);
109 int x = display.width() / 2;
110 int y = display.height() / 2;
111 display.drawString(x, y + 5, "AZWAY RETRO");
112}

120void loadingScreen()
121{
122 display.clear();
123 drawLogo();
124 int x = display.width() / 2;
125 display.setTextAlignment(TEXT_ALIGN_CENTER);
126 display.setFont(ArialMT_Plain_10);
127 display.drawString(x, 52, "Only Retrogaming");
128 display.display();
129 delay(10000);
130 display.clear();
131 display.display();
132 currentStatus = WAITING;
133}

154void mainScreen()
155{
156 topFrame();
157 display.drawXbm(32, 11, bmpStatus_width, bmpStatus_height, bmpStatus);
158 if (currentStatus == READY)
159 {
160 // Use white stars
161 display.drawXbm(6, 8, bmpStar_width, bmpStar_height, bmpStar[1]);
162 display.drawXbm(106, 8, bmpStar_width, bmpStar_height, bmpStar[1]);
163 }
164 else
165 {
166 // Use black stars
167 display.drawXbm(6, 8, bmpStar_width, bmpStar_height, bmpStar[0]);
168 display.drawXbm(106, 8, bmpStar_width, bmpStar_height, bmpStar[0]);
169 }
170}
