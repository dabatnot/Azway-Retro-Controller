104void manageLED(void *pvParameters)
105{
106 while (1)
107 {
108 switch (currentStatus)
109 {
110 case OFF:
111 digitalWrite(ledPin, LOW);
112 vTaskDelay(100 / portTICK_PERIOD_MS); // Check every 100 ms
113 break;
114 case READY:
115 digitalWrite(ledPin, HIGH);
116 vTaskDelay(100 / portTICK_PERIOD_MS); // Check every 100 ms
117 break;
118 case WAITING:
119 digitalWrite(ledPin, HIGH);
120 vTaskDelay(200 / portTICK_PERIOD_MS); // On for 200 ms
121 digitalWrite(ledPin, LOW);
122 vTaskDelay(1000 / portTICK_PERIOD_MS); // Off for 1000 ms
123 break;
124 case CONFIG:
125 digitalWrite(ledPin, HIGH);
126 vTaskDelay(25 / portTICK_PERIOD_MS); // On for 25 ms
127 digitalWrite(ledPin, LOW);
128 vTaskDelay(100 / portTICK_PERIOD_MS); // Off for 100 ms
129 break;
130 }
131 }
132}

140void activateLeds(int nbPlayers, bool showStatus)
141{
142 if (nbPlayers > 4)
143 {
144 nbPlayers = 4;
145 }
146 else if (nbPlayers < 0)
147 {
148 nbPlayers = 1;
149 }
150
151 bool ledStatus[4] = {false, false, false, false};
152 for (int currentJoy = 0; currentJoy < nbPlayers; currentJoy++)
153 {
154 ledStatus[currentJoy] = true;
155 }
156 for (int currentJoy = 0; currentJoy < 4; currentJoy++)
157 {
158 if (ledStatus[currentJoy])
159 {
160 if (showStatus)
161 {
162 display.drawXbm(joyPos[currentJoy] * 32, 32, Joystick_icon_width, Joystick_icon_width, JoyON[currentJoy]);
163 }
164 }
165 else
166 {
167 if (showStatus)
168 {
169 display.drawXbm(joyPos[currentJoy] * 32, 32, Joystick_icon_width, Joystick_icon_width, JoyOFF[currentJoy]);
170 }
171 }
172 }
173}

178void setupLED()
179{
180 // Initialize pin
181 pinMode(ledPin, OUTPUT);
182 digitalWrite(ledPin, LOW);
183
184 // Initialize serial port
185 Serial.begin(115200);
186
187 // Create tasks
188 xTaskCreatePinnedToCore(
189 manageLED, // Task function
190 "Manage LED", // Task name
191 1024, // Stack size
192 NULL, // Task parameter
193 1, // Task priority
194 &Task1, // Task handle
195 1); // CPU core to run the task
196}
